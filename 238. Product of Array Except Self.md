## 238. Product of Array Except Self
Here is the link to the problem on LeetCode 242:
[238. Product of Array Except Self](https://leetcode.com/problems/product-of-array-except-self/description/)

#### Example 1:

- Input: nums = [1,2,3,4]
- ##### Output: [24,12,8,6]

Given an integer array nums, return an array answer such that answer[i] is equal to the product of all the elements of nums except nums[i].The product of any prefix or suffix of nums is guaranteed to fit in a 32-bit integer.
<br>
You must write an algorithm that runs in O(n) time and without using the division operation.


`````Java
import java.util.Arrays;

public class Main {
    public static void main(String[] args) {
        int[] nums = {2, 2, 3, 4};
        int[] result = productExceptSelf(nums);
        System.out.println(Arrays.toString(result)); // Output: [24, 12, 8, 6]
    }

    public static int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] output = new int[n];

        // Calculate left products
        int leftProduct = 1;
        for (int i = 0; i < n; i++) {
            output[i] = leftProduct;
            leftProduct *= nums[i];
        }

        // Calculate right products and multiply with left products
        int rightProduct = 1;
        for (int i = n - 1; i >= 0; i--) {
            output[i] *= rightProduct;
            rightProduct *= nums[i];
        }

        return output;
    }
}

import java.util.Arrays;

public class Main {
    public static void main(String[] args) {
        int[] nums = {2, 2, 3, 4};
        
    
        int n = nums.length;

        // Array to store all left multiplication
        int[] left = new int[n];

        // Array to store all right multiplication
        int[] right = new int[n];

        // Initialize left and right arrays
        left[0] = 1;
        for (int i = 1; i < n; i++) {
            left[i] = left[i - 1] * nums[i - 1];
        }

        right[n - 1] = 1;
        for (int i = n - 2; i >= 0; i--) {
            right[i] = right[i + 1] * nums[i + 1];
        }

        // Calculate the result by multiplying left and right products
        int[] ans = new int[n];
        for (int i = 0; i < n; i++) {
            ans[i] = left[i] * right[i];
        }

       System.out.println(Arrays.toString(ans));
    }
